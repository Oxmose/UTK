59 void bsp_clk_pll_init(void)
60 {
61  /* PLL should be disabled at that point */
62  *RCC_PLLCFGR_REGISTER = *RCC_PLLCFGR_REGISTER &
63  ~(RCC_PLLCFGR_PLLM_MASK | RCC_PLLCFGR_PLLN_MASK |
64  RCC_PLLCFGR_PLLP_MASK | RCC_PLLCFGR_PLLQ_MASK |
65  RCC_PLLCFGR_PLLSRC_MASK);
66 
67  /* Set the PLL values */
68  *RCC_PLLCFGR_REGISTER = *RCC_PLLCFGR_REGISTER |
69  (RCC_PLLM_VALUE << RCC_PLLCFGR_PLLM_OFFSET) |
70  (RCC_PLLN_VALUE << RCC_PLLCFGR_PLLN_OFFSET) |
71  (RCC_PLLP_VALUE << RCC_PLLCFGR_PLLP_OFFSET) |
72  (RCC_PLLQ_VALUE << RCC_PLLCFGR_PLLQ_OFFSET) |
73  (RCC_PLLSRC_VALUE << RCC_PLLCFGR_PLLSRC_OFFSET);
74 
75  /* Enable PLLs */
76  *RCC_CR_REGISTER = *RCC_CR_REGISTER | RCC_CR_PLLON;
77 
78  /* Wait for PLL lock */
79  while((*RCC_CR_REGISTER & RCC_CR_PLLRDY) == 0);
80 }

87 void bsp_clk_switch_to_pll(void)
88 {
89  /* Set AHB, APB1 and APB2 prescaler values */
90  *RCC_CFGR_REGISTER = *RCC_CFGR_REGISTER &
91  ~(RCC_CFGR_HRPE_MASK |
92  RCC_CFGR_PPRE1_MASK |
93  RCC_CFGR_PPRE2_MASK);
94  *RCC_CFGR_REGISTER = *RCC_CFGR_REGISTER | RCC_CFGR_HRPE_DIV1 |
95  RCC_CFGR_PPRE1_DIV2 | RCC_CFGR_PPRE2_DIV1;
96 
97  /* Set the system clock to use PLL as input */
98  *RCC_CFGR_REGISTER = (*RCC_CFGR_REGISTER & ~RCC_CFGR_SW_MASK) |
99  RCC_CFGR_SW_PLL;
100 
101  /* Wait for update */
102  while((*RCC_CFGR_REGISTER & RCC_CFGR_SWS_MASK) != RCC_CFGR_SWS_PLL);
103 }

113 static uint32_t bsp_clock_get_pll_freq(void)
114 {
115  uint32_t pll_freq;
116 
117  /* Get Source clock */
118  if((*RCC_PLLCFGR_REGISTER & RCC_PLLCFGR_PLLSRC_MASK) == 0)
119  {
120  pll_freq = RCC_HSI_BASE_FREQ;
121  }
122  else
123  {
124  pll_freq = RCC_HSE_BASE_FREQ;
125  }
126 
127  /* Get PLLM value */
128  pll_freq /= (*RCC_PLLCFGR_REGISTER & RCC_PLLCFGR_PLLM_MASK) >>
129  RCC_PLLCFGR_PLLM_OFFSET;
130  /* Get PLLN value */
131  pll_freq *= (*RCC_PLLCFGR_REGISTER & RCC_PLLCFGR_PLLN_MASK) >>
132  RCC_PLLCFGR_PLLN_OFFSET;
133  /* Get PLLP value */
134  pll_freq /= (((*RCC_PLLCFGR_REGISTER & RCC_PLLCFGR_PLLP_MASK) >>
135  RCC_PLLCFGR_PLLP_OFFSET) + 1) * 2;
136 
137  return pll_freq;
138 }

188  KERNEL_LOG_INFO("GPIO clock enabled",

242 ERROR_CODE_E bsp_clk_sys_init(void)
243 {
244  ERROR_CODE_E error;
245 
246  /* Check init state */
247  if(bsp_clk_init != 0)
248  {
249 #if KERNEL_LOG_LEVEL == ERROR_LOG_LEVEL
250  KERNEL_LOG_ERROR("System clock is already initialized",
251  ERROR_ALREADY_INIT,
252  sizeof(ERROR_ALREADY_INIT),
253  ERROR_ALREADY_INIT);
254 #endif
255  return ERROR_ALREADY_INIT;
256  }
257 
258  /* Enable the power interface clock and system clock controller */
259  *RCC_APB1ENR_REGISTER = *RCC_APB1ENR_REGISTER | RCC_APB1ENR_PWREN;
260  while((*RCC_APB1ENR_REGISTER & RCC_APB1ENR_PWREN) == 0);
261  *RCC_APB2ENR_REGISTER = *RCC_APB2ENR_REGISTER | RCC_APB2ENR_SYSCFGEN;
262  while((*RCC_APB2ENR_REGISTER & RCC_APB2ENR_SYSCFGEN) == 0);
263 
264  /* Update voltage scaling */
265  error = bsp_pwr_set_scaling(POWER_SCALING_MODE_2);
266  if(error != NO_ERROR)
267  {
268  return error;
269  }
270 
271  /* Set flash latency */
272  error = bsp_flash_set_latency(FLASH_LATENCY_WS_2);
273  if(error != NO_ERROR)
274  {
275  return error;
276  }
277 
278  /* Init systems clocks */
279  bsp_clk_base_sys_init();
280  bsp_clk_pll_init();
281  bsp_clk_switch_to_pll();
282 
283  /* Init GPIO and USART clocks */
284  error = bsp_clk_gpio_enable(GPIO_ID_A);
285  if(error != NO_ERROR)
286  {
287  return error;
288  }
289  error = bsp_clk_usart_enable(USART_ID_2);
290  if(error != NO_ERROR)
291  {
292  return error;
293  }
294 
295 #if KERNEL_LOG_LEVEL >= INFO_LOG_LEVEL
296  KERNEL_LOG_INFO("System clock initialized",
297  NO_ERROR,
298  sizeof(NO_ERROR),
299  NO_ERROR);
300 #endif
301 
302  /* Update init state */
303  bsp_clk_init = 1;
304  return NO_ERROR;
305 }

307 uint32_t bsp_clocks_get_freq(const CLOCK_IDENTIFIER_T clk_id)
308 {
309  uint32_t sysclk;
310  uint32_t factor;
311 
312  /* Get Source clock */
313  if((*RCC_CFGR_REGISTER & RCC_CFGR_SWS_MASK) == RCC_CFGR_SWS_PLL)
314  {
315  sysclk = bsp_clock_get_pll_freq();
316  }
317  else if((*RCC_CFGR_REGISTER & RCC_CFGR_SWS_MASK) == RCC_CFGR_SWS_HSI)
318  {
319  sysclk = RCC_HSI_BASE_FREQ;
320  }
321  else if((*RCC_CFGR_REGISTER & RCC_CFGR_SWS_MASK) == RCC_CFGR_SWS_HSE)
322  {
323  sysclk = RCC_HSE_BASE_FREQ;
324  }
325  else
326  {
327  /* We were not able to detect the input clock */
328  return 0;
329  }
330 
331  /* Divide by AHB prescaler */
332  if((*RCC_CFGR_REGISTER & RCC_CFGR_HPRE_EN) != 0)
333  {
334  factor = (*RCC_CFGR_REGISTER & RCC_CFGR_HPRE_VAL_MASK) >>
335  RCC_CFGR_HRPE_OFFSET;
336  sysclk /= (uint32_t)bsp_hpre_factor_lookup[factor];
337  }
338 
339  if(clk_id == BSP_CLOCK_ID_HCLK || clk_id == BSP_CLOCK_ID_FCLK)
340  {
341  return sysclk;
342  }
343  else if(clk_id == BSP_CLOCK_ID_PCLK1 || clk_id == BSP_CLOCK_ID_APB1_TIMER)
344  {
345  /* Get the APB1 prescaler */
346  if((*RCC_CFGR_REGISTER & RCC_CFGR_PPRE1_EN) != 0)
347  {
348  factor = (*RCC_CFGR_REGISTER & RCC_CFGR_PPRE1_VAL_MASK) >>
349  RCC_CFGR_PPRE1_OFFSET;
350 
351  sysclk /= (uint32_t)bsp_ppre_factor_lookup[factor];
352 
353  if(clk_id == BSP_CLOCK_ID_APB1_TIMER)
354  {
355  sysclk *= 2;
356  }
357  }
358  }
359  else if(clk_id == BSP_CLOCK_ID_PCLK2 || clk_id == BSP_CLOCK_ID_APB2_TIMER)
360  {
361  /* Get the APB2 prescaler */
362  if((*RCC_CFGR_REGISTER & RCC_CFGR_PPRE2_EN) != 0)
363  {
364  factor = (*RCC_CFGR_REGISTER & RCC_CFGR_PPRE2_VAL_MASK) >>
365  RCC_CFGR_PPRE2_OFFSET;
366  sysclk /= bsp_ppre_factor_lookup[factor];
367 
368  if(clk_id == BSP_CLOCK_ID_APB2_TIMER)
369  {
370  sysclk *= 2;
371  }
372  }
373  }
374  else
375  {
376  sysclk = 0;
377  }
378 
379 #if KERNEL_LOG_LEVEL >= INFO_LOG_LEVEL
380  KERNEL_LOG_INFO("System clock frequency requested",
381  (void*)(uint32_t)sysclk,
382  sizeof(sysclk),
383  NO_ERROR);
384 #endif
385 
386  return sysclk;
387 }
