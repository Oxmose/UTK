192 ERROR_CODE_E bsp_gpio_init(const GPIO_IDENTIFIER_E gpio_id,
193  const GPIO_SETTINGS_T* settings)
194 {
195  volatile uint32_t* current_reg;
196 
197  uint8_t i;
198 
199  /* Check parameters */
200  if(settings == NULL)
201  {
202 #if KERNEL_LOG_LEVEL == ERROR_LOG_LEVEL
203  KERNEL_LOG_ERROR("GPIO settings structure is NULL",
204  ERROR_NULL_POINTER,
205  sizeof(ERROR_NULL_POINTER),
206  ERROR_NULL_POINTER);
207 #endif
208  return ERROR_NULL_POINTER;
209  }
210  CHECK_MODETYPE(settings->io_modetype);
211  CHECK_SPEED(settings->io_speed);
212  CHECK_ALTFUNC(settings->io_altfunc);
213 
214  /* Check which pin is to be configured */
215  for(i = 0; i < GPIO_PIN_COUNT; ++i)
216  {
217  /* The pin is set to be configured */
218  if(((1 << i) & settings->io_pin) != 0)
219  {
220  if((settings->io_modetype & GPIO_MODE_MASK) == GPIO_MODE_OUTPUT ||
221  (settings->io_modetype & GPIO_MODE_MASK) == GPIO_MODE_ALTFUN)
222  {
223  /* Set IO output type */
224  current_reg = GPIO_X_REGISTER(gpio_id, GPIO_OTYPER_OFFSET);
225  *current_reg = (*current_reg & ~GPIO_OTYPER_MASK(i)) |
226  (((settings->io_modetype & GPIO_MODE_MASK)
227  >> GPIO_MODE_OFFSET)
228  << i);
229  /* Set IO output speed */
230  current_reg = GPIO_X_REGISTER(gpio_id, GPIO_OSPEEDR_OFFSET);
231  *current_reg = (*current_reg & ~GPIO_OSPEEDR_MASK(i)) |
232  (settings->io_speed << (i * 2));
233  }
234 
235  if((settings->io_modetype & GPIO_MODE_MASK) == GPIO_MODE_ALTFUN)
236  {
237  /* Set the alternate function */
238  if(i > 7)
239  {
240  current_reg = GPIO_X_REGISTER(gpio_id, GPIO_AFRH_OFFSET);
241  }
242  else
243  {
244  current_reg = GPIO_X_REGISTER(gpio_id, GPIO_AFRL_OFFSET);
245  }
246 
247  *current_reg = (*current_reg & ~GPIO_AFR_MASK(i)) |
248  (settings->io_altfunc << ((i & 0x7) * 4));
249  }
250 
251  /* Set the GPIO mode */
252  current_reg = GPIO_X_REGISTER(gpio_id, GPIO_MODER_OFFSET);
253  *current_reg = (*current_reg & ~GPIO_MODER_MASK(i)) |
254  (settings->io_modetype << (i * 2));
255 
256  /* Set the PU/PD behavior */
257  current_reg = GPIO_X_REGISTER(gpio_id, GPIO_PUDR_OFFSET);
258  *current_reg = (*current_reg & ~GPIO_PUPDR_MASK(i)) |
259  (((settings->io_modetype & GPIO_PUPD_MASK) >>
260  GPIO_PUPD_OFFSET)
261  << i * 2);
262  }
263  }
264 
265 #if KERNEL_LOG_LEVEL >= INFO_LOG_LEVEL
266  KERNEL_LOG_INFO("GPIO initialized",
267  (void*)(uint32_t)gpio_id,
268  sizeof(gpio_id),
269  NO_ERROR);
270 #endif
271 
272  return NO_ERROR;
273 }
