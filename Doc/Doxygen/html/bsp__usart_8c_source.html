233 void bsp_uart_init(const USART_IDENTIFIER_E usart_id,
234  const SERIAL_SETTINGS_T* settings)
235 {
236  volatile uint32_t* cr1;
237  volatile uint32_t* cr2;
238  volatile uint32_t* cr3;
239  volatile uint32_t* brr;
240 
241  uint32_t freq;
242 
243  switch(usart_id)
244  {
245  case USART_ID_1:
246  cr1 = (volatile uint32_t*)USART1_CR1_ADDRESS;
247  cr2 = (volatile uint32_t*)USART1_CR2_ADDRESS;
248  cr3 = (volatile uint32_t*)USART1_CR3_ADDRESS;
249  brr = (volatile uint32_t*)USART1_BRR_ADDRESS;
250  break;
251  case USART_ID_2:
252  cr1 = (volatile uint32_t*)USART2_CR1_ADDRESS;
253  cr2 = (volatile uint32_t*)USART2_CR2_ADDRESS;
254  cr3 = (volatile uint32_t*)USART2_CR3_ADDRESS;
255  brr = (volatile uint32_t*)USART2_BRR_ADDRESS;
256  break;
257  case USART_ID_6:
258  cr1 = (volatile uint32_t*)USART6_CR1_ADDRESS;
259  cr2 = (volatile uint32_t*)USART6_CR2_ADDRESS;
260  cr3 = (volatile uint32_t*)USART6_CR3_ADDRESS;
261  brr = (volatile uint32_t*)USART6_BRR_ADDRESS;
262  break;
263  default:
264  return;
265  }
266 
267  /* Set CR1 register */
268  *cr1 = *cr1 & ~(USART_CR1_OVER8 | USART_CR1_M |
269  USART_CR1_WAKE | USART_CR1_PCE |
270  USART_CR1_PS | USART_CR1_PEIE |
271  USART_CR1_TXEIE | USART_CR1_TCIE |
272  USART_CR1_RXNEIE | USART_CR1_IDLEIE |
273  USART_CR1_TE | USART_CR1_RE |
274  USART_CR1_RWU | USART_CR1_SBK);
275 
276  *cr1 = *cr1 |
277  (settings->word_length == 8 ? 0 : USART_CR1_M) |
278  ((settings->partity == SERIAL_PARITY_EVEN) ||
279  (settings->partity == SERIAL_PARITY_NONE) ? 0 : USART_CR1_PS) |
280  (settings->partity == SERIAL_PARITY_NONE ? 0 : USART_CR1_PCE) |
281  USART_CR1_TE | USART_CR1_RE;
282 
283  /* Set CR2 register */
284  *cr2 = *cr2 & ~(USART_CR2_LINEN | USART_CR2_STOP_MASK |
285  USART_CR2_CLKEN | USART_CR2_CPOL |
286  USART_CR2_CPHA | USART_CR2_LBCL |
287  USART_CR2_LBIE | USART_CR2_LBL |
288  USART_CR2_ADDR_MASK);
289 
290  switch(settings->stop_bits)
291  {
292  case SERIAL_STOP_BITS_05:
293  *cr2 = *cr2 | USART_CR2_STOP_05;
294  break;
295  case SERIAL_STOP_BITS_15:
296  *cr2 = *cr2 | USART_CR2_STOP_15;
297  break;
298  case SERIAL_STOP_BITS_2:
299  *cr2 = *cr2 | USART_CR2_STOP_2;
300  break;
301  case SERIAL_STOP_BITS_1:
302  default:
303  break;
304  }
305 
306  /* Set CR3 register */
307  *cr3 = *cr3 & ~(USART_CR3_ONEBIT | USART_CR3_CTSIE |
308  USART_CR3_CTSE | USART_CR3_RTSE |
309  USART_CR3_DMAT | USART_CR3_DMAR |
310  USART_CR3_SCEN | USART_CR3_NACK |
311  USART_CR3_HDSEL | USART_CR3_IRLP |
312  USART_CR3_IREN | USART_CR3_EIE);
313 
314  switch(settings->ctrl_flow)
315  {
316  case SERIAL_CTRL_FLOW_CTS:
317  *cr3 = *cr3 | USART_CR3_CTSE;
318  break;
319  case SERIAL_CTRL_FLOW_RTS:
320  *cr3 = *cr3 | USART_CR3_RTSE;
321  break;
322  case SERIAL_CTRL_FLOW_CTS_RTS:
323  *cr3 = *cr3 | USART_CR3_CTSE | USART_CR3_RTSE;
324  break;
325  case SERIAL_STOP_BITS_1:
326  default:
327  break;
328  }
329 
330 
331  /* Set BRR register */
332  if(usart_id == USART_ID_1 || usart_id == USART_ID_6)
333  {
334  freq = bsp_clocks_get_freq(BSP_CLOCK_ID_PCLK2);
335  }
336  else
337  {
338  freq = bsp_clocks_get_freq(BSP_CLOCK_ID_PCLK1);
339  }
340 
341  uint32_t mantissa = USART_GET_BRR_DIV_MANT_16(freq, settings->baudrate);
342  uint32_t fraq = USART_GET_BRR_DIV_FRAQ_16(freq, settings->baudrate);
343 
344  (void)fraq;
345  (void)mantissa;
346 
347  *brr = (USART_GET_BRR_DIV_MANT_16(freq, settings->baudrate) <<
348  USART_BRR_DIV_MANT_OFFSET) |
349  USART_GET_BRR_DIV_FRAQ_16(freq, settings->baudrate);
350 }

406 ERROR_CODE_E serial_init(const SERIAL_SETTINGS_T* settings)
407 {
412  GPIO_SETTINGS_T gpio_settings;
413  ERROR_CODE_E error;
414 
415  if(usart_init_state != 0)
416  {
417 #if KERNEL_LOG_LEVEL == ERROR_LOG_LEVEL
418  KERNEL_LOG_ERROR("Serial already initialized",
419  ERROR_ALREADY_INIT,
420  sizeof(ERROR_ALREADY_INIT),
421  ERROR_ALREADY_INIT);
422 #endif
423  return ERROR_ALREADY_INIT;
424  }
425 
426  /* Check parameters */
427  if(settings == NULL)
428  {
429 #if KERNEL_LOG_LEVEL == ERROR_LOG_LEVEL
430  KERNEL_LOG_ERROR("Serial settings structure is NULL",
431  ERROR_NULL_POINTER,
432  sizeof(ERROR_NULL_POINTER),
433  ERROR_NULL_POINTER);
434 #endif
435  return ERROR_NULL_POINTER;
436  }
437  CHECK_BAUDRATE(settings->baudrate);
438  CHECK_WDLENGTH(settings->word_length);
439  CHECK_STOPBITS(settings->stop_bits);
440  CHECK_PARITYVA(settings->partity);
441  CHECK_CTRLFLOW(settings->ctrl_flow);
442 
443  /* Init GPIO and USART clock */
444  error = bsp_clk_gpio_enable(GPIO_ID_A);
445  if(error != NO_ERROR)
446  {
447  return error;
448  }
449  error = bsp_clk_usart_enable(USART_ID_2);
450  if(error != NO_ERROR)
451  {
452  return error;
453  }
454 
455  /* Initialize the USART2 GPIO A */
456  gpio_settings.io_pin = GPIO_PIN_2 |
457  GPIO_PIN_3;
458  gpio_settings.io_modetype = GPIO_MODE_ALTFUN |
459  GPIO_TYPE_PUSHPULL |
460  GPIO_PUPD_NONE;
461  gpio_settings.io_altfunc = GPIO_ALFUNC_7;
462  gpio_settings.io_speed = GPIO_SPEED_VERY_HIGH;
463 
464  error = bsp_gpio_init(GPIO_ID_A, &gpio_settings);
465  if(error != NO_ERROR)
466  {
467  return error;
468  }
469 
470  /* Make sure USART2 is disabled */
471  bsp_usart_disable(USART_ID_2);
472 
473  /* Set USART2 settings */
474  bsp_uart_init(USART_ID_2, settings);
475 
476  /* Enable USART2 */
477  bsp_usart_enable(USART_ID_2);
478 
479 #if KERNEL_LOG_LEVEL >= INFO_LOG_LEVEL
480  KERNEL_LOG_INFO("USART initialized",
481  NO_ERROR,
482  sizeof(NO_ERROR),
483  NO_ERROR);
484 #endif
485  usart_init_state = 1;
486  return NO_ERROR;
487 }

489 ERROR_CODE_E serial_write(const char* str, const size_t length)
490 {
491  size_t i;
492 
493  if(usart_init_state == 0)
494  {
495 #if KERNEL_LOG_LEVEL == ERROR_LOG_LEVEL
496  KERNEL_LOG_ERROR("Serial needs to be initialized",
497  ERROR_NEED_INIT,
498  sizeof(ERROR_NEED_INIT),
499  ERROR_NEED_INIT);
500 #endif
501  return ERROR_ALREADY_INIT;
502  }
503 
504  for(i = 0; i < length; ++i)
505  {
506  /* Wait for serial line to be ready to transmit */
507  while((*USART2_SR_REGISTER & USART_SR_TXE) == 0);
508 
509  *USART2_DR_REGISTER = *str++;
510  }
511 
512  return NO_ERROR;
513 }
