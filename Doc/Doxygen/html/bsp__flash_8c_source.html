48 ERROR_CODE_E bsp_flash_set_latency(const FLASH_LATENCY_WS_E latency)
49 {
50  /* Sets the flash latency value, we don't apply mask verification because
51  * the FLASH_LATENCY_WS_E enumeration should already contain the right
52  * values.
53  */
54  *FLASH_ACR_REGISTER = (*FLASH_ACR_REGISTER & ~FLASH_ACR_LATENCY_MASK) |
55  latency;
56 
57  /* Delay to ensure bit update */
58  while((*FLASH_ACR_REGISTER & FLASH_ACR_LATENCY_MASK) != latency);
59 
60 #if KERNEL_LOG_LEVEL >= INFO_LOG_LEVEL
61  KERNEL_LOG_INFO("Flash lantency changed",
