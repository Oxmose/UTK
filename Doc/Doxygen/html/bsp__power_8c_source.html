33 ERROR_CODE_E bsp_pwr_set_scaling(const POWER_SCALING_E pwr_scaling)
34 {
35  /* Sets the power scaling value, we don't apply mask verification because
36  * the POWER_SCALING_E enumeration should already contain the right values.
37  */
38  *PWR_CR_REGISTER = (*PWR_CR_REGISTER & ~PWR_CR_VOS_MASK) | pwr_scaling;
39 
40  /* Delay to ensure bit update */
41  while((*PWR_CR_REGISTER & PWR_CR_VOS_MASK) != pwr_scaling);
42 
43 #if KERNEL_LOG_LEVEL >= INFO_LOG_LEVEL
44  KERNEL_LOG_INFO("Power scaling mode changed",

49
